## 8888 '시험'

- 문제의 조건들이 복잡했지만, 이를 cmp 함수에 잘 적용하여 sort만 하면 쉽게 해결 가능하다.


```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
using namespace std;

/*
각 테스트 케이스의 첫 번째 줄에는 N, T, P가 공백으로 구분되어 주어진다.
(1 ≤ P ≤ N ≤ 2000, 1 ≤ T ≤ 2000)
 * */

class person{
public:
    int num;
    int score;
    int solve;
};


int tc;
int a[2001][2001];
int problem[2001]; // 문제 점수


// 등수 체크
bool cmp(person p1, person p2){
    if(p1.score == p2.score){
        if(p1.solve == p2.solve){
            return p1.num < p2.num;
        }else{
            return p1.solve > p2.solve;
        }
    }else{
        return p1.score > p2.score;
    }
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    cin >> tc;
    int casenum = 1;
    while(tc--){
        int n, t, p, ans = 0;
        cin >> n >> t >> p;
        vector<person> info (n+1);
        memset(problem, 0, sizeof(problem));
        memset(a, 0, sizeof(a));

        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= t; j++){
                cin >> a[i][j];
                if(a[i][j] == 0){
                    problem[j]++;
                }
            }
        }

        // 점수 체크
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= t; j++){
                // 참가자별 획득 점수
                info[i].num = i;

                if(a[i][j] == 1) {
                    info[i].score += problem[j];
                    info[i].solve ++;
                }
            }
        }

        sort(info.begin() + 1, info.end(), cmp);

        // p의 인덱스 찾기
        for(int i = 1; i <= n; i++){
            if(info[i].num == p){
                //whereP = i;
                cout << "#" << casenum++ << " " << info[i].score << " " << i << '\n';
                break;
            }
        }
    }

    return 0;
}
```
