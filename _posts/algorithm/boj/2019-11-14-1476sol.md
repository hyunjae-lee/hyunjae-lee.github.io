## 1476 날짜 계산

- 최대 15의 E * 최대 28의 S * 최대 19의 M = 7980 개의 경우의 수
- 전부 다 한다.
  
```cpp
  // problem url : https://www.acmicpc.net/problem/1476
#include <iostream>
using namespace std;

int main(){

    int e, s, m;
    cin >> e >> s >> m;
    int E = 1, S =1 , M = 1;

    for(int i = 1; ; i++){
        if(E == e && S == s && M == m){
            cout << i << "\n";
            break;
        }
        E++;
        S++;
        M++;

        if(E == 16){
            E = 1;
        }

        if(S == 29){
            S = 1;
        }

        if(M == 20){
            M = 1;
        }
    }

    return 0;
}

```
