## 17829 - 222-풀링

- 특별한 알고리즘보다는 구현할 줄 아는지를 물어보는 문제. (i,j)의 위치에서 자신을 포함하여 주변 3칸 중에서 두 번째로 큰 값을 찾는다. 그리고 두 칸씩 점프. 이게 끝이다.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#define sz 1024
using namespace std;

int n;
int a[sz][sz];

int main(){
    cin >> n;

    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            cin >> a[i][j];
        }
    }

    while(n!=1){
        vector<int> cmp(4);
        
        for(int i = 0; i < n-1; i += 2){
            for(int j = 0; j < n-1; j += 2){
                cmp[0] = a[i][j];
                cmp[1] = a[i][j+1];
                cmp[2] = a[i+1][j];
                cmp[3] = a[i+1][j+1];
                sort(cmp.begin(), cmp.end());
                a[i/2][j/2] = cmp[2];
            }
        }
        n /= 2;
    }

    cout << a[0][0] << endl;


    return 0;
}
```
