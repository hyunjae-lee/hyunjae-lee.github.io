## 17143 낚시왕

- 해결해야 할 문제들을 다음과 같이 나눠보았다.
  - 문제 포인트 1. 입력받은 상어의 상태에 따라 상어의 움직임 처리 및 맵 업데이트
  - 문제 포인트 2. 방향/속도에 따라 어떻게 상어가 위치할 것인가?
  - 문제 포인트 3. 한 위치에 여러 마리의 상어에 대한 정보는 어떻게 저장 및 처리할 것인가?

* 입력받은 상어의 상태에 따라 상어의 움직임 처리 및 맵 업데이트
  * 답변

* 방향/속도에 따라 어떻게 상어가 위치할 것인가?
  * 답변

* 한 위치에 여러 마리의 상어에 대한 정보는 어떻게 저장 및 처리할 것인가?
  * 1. 답변


```cpp
  // problem url : https://www.acmicpc.net/problem/17143
#include <iostream>
#include <vector>
#include <algorithm>
#define MAX 101
using namespace std;

int n, m, shark;

struct Shark{
    int x;
    int y;
    int size;
    int dir;
    int speed;
    bool alive;
};

bool compare(Shark a, Shark b){
    // 0,0 / 0,1 / ... / 1,1 / 1,2
    if(a.x == b.x){
        if(a.y == b.y){
            return a.size < b.size;
        }else{
            return a.y < b.y;
        }
    }else{
        return a.x < b.x;
    }
}

int a[MAX][MAX];

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> n >> m >> shark;
    
    if(shark == 0){
        cout << 0;
        return 0;
    }
    
    // 상어들의 상태를 저장할 벡터
    vector<Shark> v;
    
    for(int i = 0; i < shark; i++){
        Shark s;
        cin >> s.x >> s.y >> s.speed >> s.dir >> s.size;
        s.alive = true;
        a[s.x][s.y] += 1;
        
        v.push_back(s);
    }
    
    // 상어들의 상태를 담은 벡터를 x, y, speed 크기로 오름차순 정렬한다.
    sort(v.begin(), v.end(), compare);
    
    int ans = 0;
    
    for(int j = 1; j <= n; j++){
        for(int k = 1; k <= m; k++){
            cout << a[j][k] << ' ';
        }
        cout << '\n';
    }
    cout << "======" << '\n';
    
    // 낚시왕 움직임 시작
    for(int i = 1; i <= m; i++){
        
        // 1. 낚시왕이 오른쪽으로 한 칸 이동한다.
        int cur = i;
        
        // 2. 낚시왕이 있는 열에 있는 상어 중에서 땅과 제일 가까운 상어를 잡는다. 상어를 잡으면 격자판에서 잡은 상어가 사라진다.
        
        // 상어 검사
        for(int k = 0; k < v.size(); k++){
            // 낚시왕과 동일한 열에 위치한 가장 가까운 상어 발견
            if(v[k].alive && v[k].y == cur){
                v[k].alive = false;
                a[v[k].x][cur] = 0;
                ans += v[k].size;
                break;
            }
        }
        
        // 3. 상어가 이동한다
        for(int k = 1; k <= v.size(); k++){
            
            if(v[k].speed == 0 || !v[k].alive) {
                continue;
            }
            else{
                a[v[k].x][v[k].y] --;
                int jump = 0;
                // 위
                if(v[k].dir == 1){
                    jump = v[k].x - 1;
                    if(jump >= v[k].speed){
                        v[k].x -= jump;
                    }else{
                        // 맨 위로 이동시킨다.
                        v[k].x = 1;
                        int times = v[k].speed / (n-1);
                        if(times % 2 == 0){
                            v[k].dir = 2;
                        }
                        v[k].x += v[k].speed % (n-1);
                    }
                    
                    a[v[k].x][v[k].y] ++;
                }
                // 아래
                else if(v[k].dir == 2){
                    jump = n - v[k].x;
                    if(jump >= v[k].speed){
                        v[k].x += jump;
                    }else{
                        // 맨 아래로 이동시킨다.
                        v[k].x = n;
                        int times = v[k].speed / (n-1);
                        if(times % 2 == 0){
                            v[k].dir = 1;
                        }
                        v[k].x -= v[k].speed % (n-1);
                    }
                    
                    a[v[k].x][v[k].y] ++;
                }
                // 오른쪽
                else if(v[k].dir == 3){
                    jump = m - v[k].y;
                    if(jump >= v[k].speed){
                        v[k].y += jump;
                    }else{
                        // 맨 오른쪽으로 이동시킨다.
                        v[k].y = m;
                        int times = v[k].speed / (m-1);
                        if(times % 2 == 0){
                            v[k].dir = 4;
                        }
                        v[k].y -= v[k].speed % (m-1);
                    }
                    
                    a[v[k].x][v[k].y] ++;
                }
                // 왼쪽
                else if(v[k].dir == 4){
                    jump = v[k].y - 1;
                    if(jump >= v[k].speed){
                        v[k].y -= jump;
                    }else{
                        // 맨 왼쪽 이동시킨다.
                        v[k].y = 1;
                        int times = v[k].speed / (m-1);
                        if(times % 2 == 0){
                            v[k].dir = 3;
                        }
                        v[k].y += v[k].speed % (m-1);
                    }
                    
                    a[v[k].x][v[k].y] ++;
                }
            }
        }
        
        // 상어들의 상태를 담은 벡터를 x, y, speed 크기로 오름차순 정렬한다.
        sort(v.begin(), v.end(), compare);
        
        for(int j = 1; j <= n; j++){
            for(int k = 1; k <= m; k++){
                // 상어가 해당 위치에 두 마리 이상이다.
                if(a[j][k] >= 2){
                    int count = a[j][k] - 1; // 없앨 상어의 수
                    for(int z = 0; z < v.size(); z++){
                        if(count == 0) break;
                        if(v[z].alive &&
                           v[z].x == j && v[z].y == k && count > 0){
                            v[z].alive = false;
                            a[v[k].x][v[k].y] --;
                            count--;
                        }
                    }
                }
            }
        }
        
        for(int j = 1; j <= n; j++){
            for(int k = 1; k <= m; k++){
                cout << a[j][k] << ' ';
            }
            cout << '\n';
        }
        cout << "======" << '\n';
        
        
    }
    
    cout << ans << '\n';
    
    return 0;
}
```
